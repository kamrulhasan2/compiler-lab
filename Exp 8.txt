Ex-8: Write a simple lex specification to recognize digit

%{
#include <stdio.h>
/* digit */
%}
%%
[ \t\n]+        ;   /* ignore whitespace */
[0-9][0-9]+     { printf("%s: is not a digit\n", yytext); }
[0-9]           { printf("%s: is a digit\n", yytext); }
[a-zA-Z]+       { printf("%s: is not a digit\n", yytext); }
.               { ECHO; }
%%
int main()
{
    yylex();
    return 0;
}
