Ex-6: Write a simple lex specification to recognize positive and negative integer and float

%{
#include <stdio.h>
/* float or integer */
%}
%%
[ \t\n]+            ;   /* ignore whitespace */
[0-9]*\.[0-9]+      { printf("%s: is a positive float\n", yytext); }
-[0-9]*\.[0-9]+     { printf("%s: is a negative float\n", yytext); }
[0-9]+              { printf("%s: is a positive integer\n", yytext); }
-[0-9]+             { printf("%s: is a negative integer\n", yytext); }
[a-zA-Z]+           { printf("%s: is not a number\n", yytext); }
.                   { ECHO; }
%%
int main()
{
    yylex();
    return 0;
}
